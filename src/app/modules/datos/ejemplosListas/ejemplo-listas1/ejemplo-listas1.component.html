          namespace Ejemplo1{
              class Program{
                  static void Main(string[] args){
                      int cant;
                      Console.WriteLine("Ingrese la cantidad de datos agregar");
                      cant = Int32.Parse(Console.ReadLine());
                      if (cant <= 5){
                          int[] arreglo;
                          arreglo = new int[5];
                          int i,suma;
                          for(i = 0; i < cant; i ++){
                              Random rand = new Random();
                              arreglo[i] = rand.Next (1,10);
                          }
                          suma = 0;
                          for(i = 0; i < cant; i ++){
                              Console.WriteLine("numero: {0}",arreglo[i]);
                              suma += arreglo[i];
                          }
                          Console.WriteLine("La suma de todos los número aleatorios es: {0}",suma);
                      }else{
                          Console.WriteLine("La cantidad de numeros que desea ingresar es mayor a 5");
                      }
                  }
              }
          }
