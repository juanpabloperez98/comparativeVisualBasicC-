          namespace Ejemplo3{
              class Program{
                static void Main(string[] args){
                  int[,] matrix1 = new int[100,100];
                  int[,] matrix2 = new int[100,100];
                  int[,] matrix3 = new int[100,100];
                  int i,j,n;
                  Console.WriteLine("Ingrese el tamano de las matrices:");
                  n = Int32.Parse(Console.ReadLine());
                  for(i = 0; i < n; i ++){
                      for(j = 0; j < n; j ++){
                          Random rand = new Random();
                          matrix1[i,j] = rand.Next (1,10);
                          matrix2[i,j] = rand.Next (1,10);
                          matrix3[i,j] = matrix1[i,j] + matrix2[i,j];
                      }
                  }
                  Console.WriteLine("La matriz resultante es:");
                  for(i = 0; i < n; i ++){
                      for(j = 0; j < n; j ++){
                          Console.WriteLine("Posicion {0}{1}: {2}",i,j,matrix3[i,j]);
                      }
                  }
                }
            }
          }
