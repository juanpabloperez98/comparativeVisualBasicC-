          namespace Ejemplo1{
              class Program{
                  static void Main(string[] args){
                      int n, i,j;
                      Console.WriteLine("Ingrese la lognitud de la matriz:");
                      n = Int32.Parse(Console.ReadLine());
                      int[,] matrix = new int[n, n];
                      for (i = 0; i < n; i++){
                          for (j = 0; j < n; j++){
                              Console.WriteLine("Ingrese el valor en la posicion {0} - {1}",i,j);
                              matrix[i,j] = Int32.Parse(Console.ReadLine());
                          }
                      }
                      for (i = 0; i < n; i++){
                          for (j = 0; j < n; j++){
                              Console.WriteLine("Valor en la posicion  {0} - {1}: {2}",i,j,matrix[i,j]);
                          }
                      }
              }
            }
          }
