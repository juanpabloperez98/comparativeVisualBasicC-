          namespace Ejemplo4{
            class Program{
                static void Main(string[] args){
                    int op,valor;
                    Console.WriteLine("1)Convertir de centímetros a pulgadas 2)Convertir de líbra a kilogramos");
                    Console.WriteLine("Ingresa la opción: ");
                    op = Int32.Parse(Console.ReadLine());
                    Console.WriteLine("Ingresa valor: ");
                    valor = Int32.Parse(Console.ReadLine());
                    switch(op){
                        case 1:{
                            double pulgadas = valor/2.54;
                            Console.WriteLine("Centimetros: {0} pulgadas: {1}",(valor,pulgadas));
                            break;
                        }
                        case 2:{
                            double kilogramos = valor * (1/2.21);
                            Console.WriteLine("Libra: {0} kilogramos: {1}",(valor,kilogramos));
                            break;
                        }
                        default:{
                            Console.WriteLine("Opción no valida");
                            break;
                        }
                    }
                }
            }
        }
