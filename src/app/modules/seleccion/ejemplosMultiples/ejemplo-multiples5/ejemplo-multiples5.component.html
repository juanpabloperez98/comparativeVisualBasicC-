          namespace Ejemplo5{
            class Program{
                static void Main(string[] args){
                    int opcion;
                    Console.WriteLine("1.)Computadores - 2.)Celulares");
                    Console.WriteLine("Ingrese una categoria:");
                    opcion = Int32.Parse(Console.ReadLine());
                    switch(opcion){
                        case 1:{
                            Console.WriteLine("Computador Acer: 1.000.000 - Computador Asus: 1.600.000");
                            break;
                        }
                        case 2:{
                            Console.WriteLine("Celular iPhone 6: 1.800.000 - Celular Samsung j5: 700.000");
                            break;
                        }
                        default:{
                            Console.WriteLine("No existe esta categoria");
                            break;
                        }
                    }
                }
            }
        }
